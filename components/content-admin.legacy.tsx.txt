function formatDate(value: string) {
  if (!value) return '—';
  const date = new Date(value);
  if (Number.isNaN(date.getTime())) return value;
  return `${date.toLocaleDateString('vi-VN')} ${date.toLocaleTimeString('vi-VN', { hour: '2-digit', minute: '2-digit' })}`;
}

function labelSourceStatus(value: unknown) {
  return value === 'verified' ? 'verified' : value === 'reviewed' ? 'reviewed' : 'seed/demo';
}

function sourceBadgeClass(value: unknown) {
  if (value === 'verified') return 'border-emerald-200 bg-emerald-50 text-emerald-700';
  if (value === 'reviewed') return 'border-blue-200 bg-blue-50 text-blue-700';
  return 'border-amber-200 bg-amber-50 text-amber-800';
}

function qualityBadgeClass(value: unknown) {
  return value === 'critical'
    ? 'border-rose-200 bg-rose-50 text-rose-700'
    : 'border-amber-200 bg-amber-50 text-amber-800';
}

function getSourceMeta(item: Record<string, unknown> | null | undefined) {
  const meta = item && typeof item === 'object' ? (item as any).sourceMeta : null;
  return meta && typeof meta === 'object' ? meta as Record<string, unknown> : null;
}

    const grouped = new Map<string, { packId: string; label: string; total: number; pendingReview: number; readyForLeaderApproval: number; leaderApproved: number }>();

        label: `${item.grade} · ${item.subject} · ${item.book}`,

        pendingReview: 0,
        readyForLeaderApproval: 0,
        leaderApproved: 0

      if (item.reviewConsensus?.hasLeaderApproval) current.leaderApproved += 1;
      grouped.set(item.packId, current);
    }
    return Array.from(grouped.values()).sort((a, b) => a.label.localeCompare(b.label, 'vi'));

    const items = summary?.evidenceReviewWorkbench.items || [];

  const isAllowed = user?.role === 'leader' || user?.role === 'admin';

  useEffect(() => {
    if (!isAllowed) return;
    void Promise.all([loadSummary(), loadItems(entity), loadMemberships()]);

    if (!isAllowed || !selectedId) {
      setReviewItems([]);
      return;

    void loadReviews(entity, selectedId);
  }, [entity, selectedId, isAllowed]);

    setMembershipEditor(JSON.stringify(createDefaultMembership(user), null, 2));

  useEffect(() => {
    setPackSelectionText(JSON.stringify(createDefaultPackSelection(scope), null, 2));
  }, [scope.level, scope.grade, scope.subject]);

    if (!isAllowed) return;
    if (!evidenceReviewPackOptions.length) {
      setSelectedEvidenceReviewPackId('');

      return;

      setSelectedEvidenceReviewPackId(nextPackId);
      return;

    void loadEvidenceReviewPackDetail(nextPackId);

  }, [evidenceReviewItemsInPack, selectedEvidenceReviewId]);

      setEvidenceReviewDetail(null);

    }

      setMessage(json.error || 'Không tải được chi tiết evidence review theo pack.');
      setEvidenceReviewPackDetail(null);

    setEvidenceReviewPackDetail(json);

  async function loadEvidenceReviewDetail(item: NonNullable<typeof selectedEvidenceReviewItem>) {
    const params = new URLSearchParams({
      packId: item.packId,
      entityType: item.entityType,
      entityId: item.entityId,
      field: item.field

    const res = await fetch(`/api/admin/content/packs/evidence-review-actions?${params.toString()}`);
    const json = await res.json() as EvidenceReviewDetailResponse;
    if (!res.ok) {
      setMessage(json.error || 'Không tải được chi tiết field evidence review.');
      return;

    setEvidenceReviewDetail(json);

  async function submitEvidenceReviewAction(action: 'review' | 'leader_approval' | 'leader_revoke') {
    if (!selectedEvidenceReviewItem) {
      setMessage('Chưa chọn field evidence review để thao tác.');
      return;

    setEvidenceReviewBusy(true);
    try {
      const res = await fetch('/api/admin/content/packs/evidence-review-actions', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          action,
          packId: selectedEvidenceReviewItem.packId,
          entityType: selectedEvidenceReviewItem.entityType,
          entityId: selectedEvidenceReviewItem.entityId,
          field: selectedEvidenceReviewItem.field,
          decision: evidenceReviewDecision,
          note: evidenceReviewNote,
          citationLocatorVerified,
          legalSafeVerified
        })
      });
      const json = await res.json() as EvidenceReviewMutationResponse;
      if (!res.ok) {
        setMessage(json.error || 'Không cập nhật được evidence review workflow.');
        return;
      }
      setEvidenceReviewDetail(json);
      setSummary((current) => current ? {
        ...current,
        evidenceReviewSummary: json.summary,
        evidenceReviewWorkbench: json.workbench
      } : current);
      if (selectedEvidenceReviewPackId) {
        await loadEvidenceReviewPackDetail(selectedEvidenceReviewPackId);
      }
      setMessage(json.summaryText || 'Đã cập nhật evidence review workflow.');
      setEvidenceReviewNote('');
    } finally {
      setEvidenceReviewBusy(false);
    }
  }

  async function submitEvidenceReviewPackAction(action: 'review_missing_my_decision' | 'leader_approve_ready' | 'leader_revoke_all') {
    if (!selectedEvidenceReviewPackId) {
      setMessage('Chưa chọn pack evidence review để thao tác hàng loạt.');
      return;
    }
    setEvidenceReviewPackBusy(true);
    try {
      const res = await fetch('/api/admin/content/packs/evidence-review-pack-actions', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          action,
          packId: selectedEvidenceReviewPackId,
          decision: evidenceReviewDecision,
          note: evidenceReviewNote,
          citationLocatorVerified,
          legalSafeVerified
        })
      });
      const json = await res.json() as EvidenceReviewPackMutationResponse;
      if (!res.ok) {
        setMessage(json.error || 'Không cập nhật được evidence review stage ở cấp pack.');
        return;
      }
      setEvidenceReviewPackDetail(json);
      await loadSummary();
      setMessage([json.summaryText, json.errors.length ? `Có ${json.errors.length} lỗi/field bị bỏ qua.` : '', json.skippedFields ? `Skipped ${json.skippedFields} field.` : ''].filter(Boolean).join(' '));
      setEvidenceReviewNote('');
    } finally {
      setEvidenceReviewPackBusy(false);
    }
  }

  async function loadCurationDetail(packId: string) {
    if (!packId) {
      setCurationDetail(null);
      return;
    }
    const res = await fetch(`/api/admin/content/packs/curation-actions?packId=${encodeURIComponent(packId)}`);
    const json = await res.json() as PackCurationDetailResponse;
    if (!res.ok) {
      setMessage(json.error || 'Không tải được chi tiết pack curation.');
      setCurationDetail(null);
      return;
    }
    setCurationDetail(json);
  }

  async function submitCurationAction(action: 'review' | 'leader_approval' | 'leader_revoke' | 'admin_approval' | 'admin_revoke' | 'release_recommendation') {
    if (!selectedCurationPackId) {
      setMessage('Chưa chọn pack để thao tác curation.');
      return;
    }
    setCurationBusy(true);
    try {
      const res = await fetch('/api/admin/content/packs/curation-actions', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          action,
          packId: selectedCurationPackId,
          decision: curationDecision,
          recommendation: curationRecommendation,
          note: curationNote
        })
      });
      const json = await res.json() as PackCurationMutationResponse;
      if (!res.ok) {
        setMessage(json.error || 'Không cập nhật được pack curation.');
        return;
      }
      setCurationDetail(json);
      setSummary((current) => current ? {
        ...current,
        curationSummary: {
          ...current.curationSummary,
          total: Math.max(current.curationSummary.total, json.summary.total),
          withConflict: json.summary.withConflict,
          readyForLeaderApproval: json.summary.readyForLeaderApproval,
          readyForAdminApproval: json.summary.readyForAdminApproval,
          readyForVerifiedRelease: json.summary.readyForVerifiedRelease,
          items: (() => {
            const nextItems = current.curationSummary.items.filter((item) => item.packId !== selectedCurationPackId);
            if (json.consensus) nextItems.unshift(json.consensus);
            return nextItems;
          })()
        }
      } : current);
      setMessage(json.summaryText || 'Đã cập nhật pack curation.');
      setCurationNote('');
      await Promise.all([loadSummary(), loadReleaseManifest(selectedCurationPackId)]);
    } finally {
      setCurationBusy(false);
    }
  }

  async function loadItems(nextEntity = entity) {
    const params = new URLSearchParams({
      q: keyword,
      grade: nextEntity === 'templates' ? '' : grade,
      subject: nextEntity === 'templates' ? '' : subject,
      book: nextEntity === 'templates' ? '' : book,
      topic: nextEntity === 'templates' ? '' : topic,
      lifecycleStatus: lifecycleFilter,
      visibilityScope: visibilityFilter,
      sourceStatus: nextEntity === 'templates' ? '' : sourceStatusFilter
    });
    const res = await fetch(`/api/admin/content/${nextEntity}?${params.toString()}`);
    const json = await res.json() as EntityListResponse & { error?: string };
    if (!res.ok) {
      setMessage(json.error || 'Không tải được danh sách nội dung.');
      return;
    }
    const list = (json.items || []) as Array<Record<string, unknown>>;
    setItems(list);

      setSelectedId('');
      setEditor(JSON.stringify(createDefaultDraft(nextEntity, scope), null, 2));
      return;
    }
    selectItem(list[0]);
  }

    if (!selectedId) {

      return;

      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ id: selectedId, action })
    });
    const json = await res.json().catch(() => ({ error: 'Không chuyển được trạng thái.' })) as { error?: string; item?: Record<string, unknown> };
    if (!res.ok) {
      setMessage(json.error || 'Không chuyển được trạng thái nội dung.');
      return;
    }
    setMessage(`Đã chuyển trạng thái nội dung sang ${action}.`);
    if (json.item) setEditor(JSON.stringify(json.item, null, 2));

  }

              <div className="text-lg font-semibold text-slate-900">Lịch sử import</div>
              <div className="mt-4 space-y-3 text-sm">
                {(summary?.imports || []).length === 0 ? <div className="text-slate-600">Chưa có batch import nào.</div> : summary?.imports.map((item) => (
                  <div key={item.id} className="rounded-2xl border border-slate-200 bg-slate-50 p-3">
                    <div className="font-medium text-slate-900">{item.sourceLabel}</div>
                    <div className="text-slate-700">Target: {item.target} · Topics: {item.topics} · Templates: {item.templates}</div>
                    <div className="mt-1 text-xs text-slate-500">{item.importedBy} · {formatDate(item.createdAt)}{item.dryRun ? ' · dry-run' : ''}</div>
                  </div>
                ))}
              </div>
            </div>
          </div>
        </div>
      </div>

      <div className="grid gap-6 xl:grid-cols-[0.9fr,1.1fr]">
        <div className="rounded-3xl border border-slate-200 bg-white p-5 shadow-sm">
          <div className="mb-3 flex flex-wrap items-center justify-between gap-2">
            <div>
              <div className="text-lg font-semibold text-slate-900">Membership directory</div>
              <div className="text-sm text-slate-600">Vai trò login giờ bám theo membership active cùng trường/tổ thay vì chỉ chọn tay.</div>
            </div>
            <div className="text-xs text-slate-500">Tổng: {summary?.membershipSummary.total ?? 0} · Leader: {summary?.membershipSummary.leaders ?? 0} · Invited: {summary?.membershipSummary.invited ?? 0}</div>
          </div>
          <div className="flex gap-2">
            <input value={membershipSearch} onChange={(event: any) => setMembershipSearch(event.target.value)} placeholder="Tìm theo tên, email, vai trò..." className="w-full rounded-xl border border-slate-300 bg-white px-3 py-2" />
            <select value={membershipStatusFilter} onChange={(event: any) => setMembershipStatusFilter(event.target.value)} className="rounded-xl border border-slate-300 bg-white px-3 py-2 text-sm">
              <option value="">Tất cả status</option>
              <option value="active">active</option>
              <option value="invited">invited</option>
              <option value="inactive">inactive</option>
            </select>
            <button onClick={() => void loadMemberships()} className="rounded-xl bg-slate-900 px-4 py-2 text-white">Lọc</button>
          </div>
          <div className="mt-4 max-h-[360px] space-y-3 overflow-y-auto pr-1">
            {membershipItems.length === 0 ? <div className="rounded-2xl border border-dashed border-slate-300 bg-slate-50 p-4 text-sm text-slate-600">Chưa có membership phù hợp.</div> : membershipItems.map((item) => (
              <button key={item.id} onClick={() => { setSelectedMembershipId(item.id); setMembershipEditor(JSON.stringify(item, null, 2)); }} className={`block w-full rounded-2xl border p-4 text-left ${selectedMembershipId === item.id ? 'border-slate-900 bg-slate-100' : 'border-slate-200 bg-slate-50 hover:bg-slate-100'}`}>
                <div className="font-medium text-slate-900">{item.name}</div>
                <div className="mt-1 text-xs text-slate-600">{item.role} · {item.status}</div>
                <div className="mt-1 text-xs text-blue-700">{item.schoolName} · {item.departmentName}</div>
                <div className="mt-1 text-[11px] text-slate-500">review: {String(item.canReviewLessons)} · manage: {String(item.canManageContent)} · publish: {String(item.canPublishContent)}</div>
              </button>
            ))}
          </div>
        </div>

        <div className="rounded-3xl border border-slate-200 bg-white p-5 shadow-sm">
          <div className="mb-3 flex flex-wrap items-center justify-between gap-2">
            <div>
              <div className="text-lg font-semibold text-slate-900">Sửa membership JSON trực tiếp</div>
              <div className="text-sm text-slate-600">Cho phép dựng directory tổ/trường trước khi đẩy sang user directory production thật.</div>
            </div>
            <div className="flex flex-wrap gap-2">
              <button onClick={() => resetMembershipDraft()} className="rounded-xl border border-slate-300 bg-white px-4 py-2 text-sm">Membership mới</button>
              <button onClick={() => void saveMembership()} className="rounded-xl bg-slate-900 px-4 py-2 text-sm text-white">Lưu membership</button>
              <button onClick={() => void deleteCurrentMembership()} className="rounded-xl border border-rose-300 bg-rose-50 px-4 py-2 text-sm text-rose-700">Xóa</button>
            </div>
          </div>
          <textarea value={membershipEditor} onChange={(event: any) => setMembershipEditor(event.target.value)} className="min-h-[360px] w-full rounded-2xl border border-slate-300 bg-slate-950 p-4 font-mono text-sm text-slate-100" spellCheck={false} />
          <div className="mt-4 grid gap-3 md:grid-cols-4 text-sm">
            <div className="rounded-2xl border border-slate-200 bg-slate-50 p-3">
              <div className="text-slate-500">Active</div>
              <div className="mt-1 text-lg font-semibold text-slate-900">{summary?.membershipSummary.active ?? 0}</div>
            </div>
            <div className="rounded-2xl border border-slate-200 bg-slate-50 p-3">
              <div className="text-slate-500">Invited / inactive</div>
              <div className="mt-1 text-lg font-semibold text-slate-900">{summary?.membershipSummary.invited ?? 0} / {summary?.membershipSummary.inactive ?? 0}</div>
            </div>
            <div className="rounded-2xl border border-slate-200 bg-slate-50 p-3">
              <div className="text-slate-500">Admins</div>
              <div className="mt-1 text-lg font-semibold text-slate-900">{summary?.membershipSummary.admins ?? 0}</div>
            </div>
            <div className="rounded-2xl border border-slate-200 bg-slate-50 p-3">
              <div className="text-slate-500">Department groups</div>
              <div className="mt-1 text-lg font-semibold text-slate-900">{summary?.membershipSummary.byDepartment.length ?? 0}</div>
            </div>
          </div>
        </div>
      </div>
    </div>
  );
}
