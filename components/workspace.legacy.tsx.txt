type MetadataResponse = {
  curriculum: {
    level: string;
    grades: Record<string, {
      name: string;
      subjects: Record<string, {
        name: string;
        books: Record<string, {
          name: string;
          topics: Record<string, { title: string }>;
        }>;
      }>;
    }>;
  };
  templates: string[];
  teachingMethods: TeachingMethod[];
  teachingTechniques: TeachingTechnique[];
  summary: {
    levels: string[];
    gradeLevelMap: Record<string, string>;
    grades: number;
    subjects: number;
    books: number;
    topics: number;
    coverage: CurriculumCoverageItem[];
    packs: CurriculumPackSummary[];
    roadmap?: CurriculumRoadmapSummary;
    sourceStatusSummary?: { seed: number; reviewed: number; verified: number };
    supportSummary?: CurriculumSupportSummary;
    coverageTruth?: CoverageTruthReport;
  };
  defaultScope?: CurriculumScopeSelection | null;
  teachingPolicy?: TeachingPolicySettings;

  storage: string;
};

type ProgramResponse = {

};

type TemplatesResponse = {

};

  function writeHeaders(extra: Record<string, string> = {}) {
    return { 'Content-Type': 'application/json', 'x-csrf-token': csrfToken, ...extra };
  }

  async function handleGenerate() {
    setMessage('Đang dựng giáo án từ mẫu và dữ liệu có cấu trúc...');
    const res = await fetch('/api/template-builder', {
      method: 'POST',
      headers: writeHeaders(),
      body: JSON.stringify({
        level,
        grade,
        subject,
        book,
        topic,
        template,
        methods,
        techniques,
        allowInternalPreview,
        lessonContext: { classSize, classroomSpace, deviceAvailability, learnerLevel, durationProfile }
      })
    });
    const json = await res.json() as GenerateResponse;
    if (!res.ok || !json.bundle) {
      setGeneratePolicy(json.policy || null);
      setMessage(json.error || 'Không sinh được giáo án.');
      return;
    }
    setBundle(json.bundle);
    setLessonQuality(json.bundle.qualityChecklist || null);
    setGeneratePolicy(json.policy || null);
    setCurrentLessonGovernance(normalizeLessonGovernanceSnapshot({
      packId: topicData?.sourceMeta?.packId || json.bundle.trace.packId,
      sourceStatus: topicData?.sourceMeta?.status || json.bundle.trace.sourceStatus,
      lifecycleStatus: topicData?.governance?.lifecycleStatus || json.bundle.trace.lifecycleStatus,
      releaseTier: json.bundle.trace.releaseTier || selectedCoverage?.releaseTier || json.policy?.actualReleaseTier,
      supportLevel: json.bundle.trace.supportLevel || selectedCoverage?.supportLevel || json.policy?.actualSupportLevel,
      sourceLabel: topicData?.sourceMeta?.sourceLabel || json.bundle.trace.sourceLabel,
      sourceReferences: topicData?.sourceMeta?.references || [],
      supportFlags: selectedCoverage?.supportFlags,
      supportNotes: selectedCoverage?.supportNotes,
      generatedAt: new Date().toISOString(),
      referenceCount: json.bundle.trace.referenceCount,
      fieldEvidenceCount: json.bundle.trace.fieldEvidenceCount,
      conflicts: json.bundle.trace.conflicts
    }));

  async function handleExport(format: 'docx' | 'pdf') {
    if (!user) {
      setMessage("Hãy đăng nhập workspace trước khi xuất " + format.toUpperCase() + ".");
      return;
    }
    const res = await fetch(`/api/export/${format}`, {
      method: 'POST',
      headers: writeHeaders(),
      body: JSON.stringify({
        title: `${subject} - ${topic}`,
        level,
        grade,
        subject,
        book,
        topic,
        template,
        methods,
        techniques,
        content: editor,
        governanceSnapshot: derivedCurrentGovernance,
        packId: derivedCurrentGovernance?.packId,
        sourceStatus: derivedCurrentGovernance?.sourceStatus || 'seed',
        lifecycleStatus: derivedCurrentGovernance?.lifecycleStatus || 'draft',

        lessonStatus: currentLessonStatus,
        sourceLabel: derivedCurrentGovernance?.sourceLabel || 'Biên tập nội bộ',
        sourceReferences: derivedCurrentGovernance?.sourceReferences || []
      })
    });
    if (!res.ok) {
      const errorJson = await res.json().catch(() => ({} as { error?: string }));
      setMessage(errorJson.error || `Không xuất được ${format.toUpperCase()}.`);
      return;
    }
    const blob = await res.blob();
    const url = window.URL.createObjectURL(blob);
    const anchor = document.createElement('a');
    anchor.href = url;
    const disposition = res.headers.get('Content-Disposition') || '';
    const matchedName = disposition.match(/filename="?([^";]+)"?/i)?.[1];
    anchor.download = matchedName || `noi-bo-${subject}-${topic}.${format}`;
    document.body.appendChild(anchor);
    anchor.click();
    anchor.remove();
    window.URL.revokeObjectURL(url);
    setMessage(`Đã xuất ${format.toUpperCase()} theo export guard và quota operating hiện tại.`);
    void loadOperatingFoundation();
  }

  async function handleSave(status: 'draft' | 'review' = 'draft') {
    if (!user) {
      setMessage('Hãy đăng nhập workspace trước khi lưu.');
      return;
    }
    setSaving(true);
    const finalStatus = status === 'review' && settings?.approvalRequired === false ? 'approved' : status;
    const payload = {
      title: `${subject} - ${topic}`,
      level,
      grade,
      subject,
      book,
      topic,
      template,
      methods,
      techniques,
      content: editor,
      status: finalStatus,
      governanceSnapshot: derivedCurrentGovernance,
      changeSummary: currentLessonId ? 'Cập nhật nội dung từ trình soạn.' : 'Tạo giáo án từ trình soạn.'
    };
    const res = await fetch(currentLessonId ? `/api/lessons/${currentLessonId}` : '/api/lessons', {
      method: currentLessonId ? 'PUT' : 'POST',
      headers: writeHeaders(),
      body: JSON.stringify(payload)
    });
    const json = await res.json() as { item?: SavedLessonPlan; error?: string; governance?: LessonGovernanceAssessment; qualityChecklist?: LessonQualityChecklistReport };
    setSaving(false);
    if (!res.ok || !json.item) {
      if (json.qualityChecklist) setLessonQuality(json.qualityChecklist);
      const governanceNote = json.governance?.issues?.filter((item) => item.severity === 'blocker').map((item) => item.message).join(' ');
      setMessage(governanceNote || json.error || 'Không lưu được giáo án.');
      return;
    }
    setCurrentLessonId(json.item.id);
    setCurrentLessonVersion(json.item.currentVersion || 1);
    setCurrentLessonStatus(json.item.status);
    setCurrentLessonGovernance(json.item.governanceSnapshot || derivedCurrentGovernance);
    setLessonQuality(json.qualityChecklist || lessonQuality);
    const actualStatus = json.item.status;
    setMessage(actualStatus === 'approved' ? `Đã lưu và duyệt. Phiên bản hiện tại: v${json.item.currentVersion}.` : actualStatus === 'review' ? `Đã gửi duyệt giáo án. Phiên bản hiện tại: v${json.item.currentVersion}.` : `Đã lưu bản nháp. Phiên bản hiện tại: v${json.item.currentVersion}.`);
    if (activeTab === 'kho') void loadLibrary();
    void loadDashboard();
    void loadOperatingFoundation();
  }

  async function handleLogin() {
    const endpoint = authMode === 'register' ? '/api/auth/register' : '/api/auth/login';
    const res = await fetch(endpoint, {

      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ name: loginName, email: loginEmail, password: loginPassword, role: loginRole, schoolName: loginSchool, departmentName: loginDepartment, inviteCode: loginInviteCode })
    });

    if (!res.ok || !json.user) {
      setMessage(json.error || (authMode === 'register' ? 'Không đăng ký được tài khoản.' : 'Không đăng nhập được vào workspace.'));
      return;
    }
    if (json.csrfToken) setCsrfToken(json.csrfToken);
    setUser(json.user);
    void loadOperatingFoundation();
    const note = json.notice ? ' ' + json.notice : json.inviteWarning ? ' ' + json.inviteWarning : json.downgraded ? ' Yêu cầu leader/admin không có invite/membership hợp lệ nên đã hạ về teacher.' : json.autoProvisioned ? ' Đã auto-provision membership teacher cho lần vào đầu.' : '';
    setMessage(`${authMode === 'register' ? 'Đã đăng ký và vào workspace' : 'Đã vào workspace'} với vai trò ${labelRole(json.user.role)} · ${json.user.schoolName} / ${json.user.departmentName}.${note}`);
  }

                        </div>
                        {demoReadiness.blockers.length ? (
                          <div className="rounded-xl border border-amber-200 bg-white p-3 text-xs text-amber-800">
                            <div className="font-medium">Blocker demo/runtime còn lại</div>
                            <ul className="mt-2 list-disc space-y-1 pl-5">
                              {demoReadiness.blockers.slice(0, 4).map((item) => <li key={item.code}>{item.label}: {item.detail}</li>)}

                          </div>
                        ) : <div className="rounded-xl border border-emerald-200 bg-white p-3 text-xs text-emerald-800">Không có blocker source-level trong demo readiness. Vẫn phải chạy typecheck/build/live smoke thật trước khi public test rộng.</div>}
                      </div>
                    ) : <EmptyState label="Chưa tải được demo readiness board." />}
                  </div>

                  {operatingEntitlement ? (
                    <div className="rounded-2xl border border-emerald-200 bg-emerald-50 p-4">
                      <div className="font-medium text-emerald-950">Runtime quota hiện tại</div>
                      <div className="mt-1 text-sm text-emerald-900">User: {operatingEntitlement.userName} · Plan: {operatingEntitlement.plan.name} · {operatingEntitlement.assignmentSource === 'assignment' ? 'có assignment' : 'fallback Free'}</div>
                      <div className="mt-3 grid gap-3 sm:grid-cols-2 xl:grid-cols-4">
                        <MiniMetric label="Saved lessons" value={`${operatingEntitlement.savedLessons.used}/${operatingEntitlement.savedLessons.limit}`} />
                        {operatingEntitlement.usage.slice(0, 3).map((metric) => (
                          <MiniMetric key={metric.action} label={metric.action} value={metric.limit === null ? metric.used : `${metric.used}/${metric.limit}`} />
                        ))}
                      </div>
                      <div className="mt-3 text-xs text-emerald-800">{operatingEntitlement.watermarkRequired ? 'Plan hiện tại yêu cầu watermark khi export.' : 'Plan hiện tại không tự yêu cầu watermark, nhưng export guard/compliance vẫn có thể bắt watermark.'}</div>
                    </div>
                  ) : null}

                  <div className="grid gap-4 md:grid-cols-2 xl:grid-cols-4">
                    <MiniStat label="Subject slots 1–12" value={String(operatingBoard.coverageAudit.totalRequiredSubjectSlots)} />
                    <MiniStat label="Phủ bề mặt" value={operatingBoard.coverageAudit.surfaceCoveragePercent + '%'} />
                    <MiniStat label="Truth readiness" value={operatingBoard.coverageAudit.truthReadinessPercent + '%'} />
                    <MiniStat label="Verified scopes" value={String(operatingBoard.coverageAudit.verifiedScopes)} />
                    <MiniStat label="Operational scopes" value={String(operatingBoard.coverageAudit.operationalScopes)} />
                    <MiniStat label="Saved lessons" value={String(operatingBoard.lessonBiasAudit.totalSavedLessons)} />
                    <MiniStat label="Lớp 6 share" value={operatingBoard.lessonBiasAudit.grade6SharePercent + '%'} />
                    <MiniStat label="Community public" value={String(operatingBoard.communityModeration.public)} />
                    <MiniStat label="Activity games" value={String(operatingBoard.activityGameLibrary?.total || 0)} />
                    <MiniStat label="Creator profiles" value={String(operatingBoard.communityHub?.creatorProfiles || communityHub?.summary.creatorProfiles || 0)} />
                    <MiniStat label="Forum queue" value={String(operatingBoard.forumThreadSafety?.moderationQueue || 0)} />
                  </div>

                  {operatingBoard.coverageAudit.truthWarnings.length || operatingBoard.coverageAudit.priorityActions.length ? (
                    <div className="rounded-2xl border border-amber-200 bg-amber-50 p-4 text-sm">
                      <div className="font-semibold text-amber-950">Coverage truth & bias guard</div>
                      <div className="mt-2 text-amber-900">Bảng này phân biệt coverage bề mặt với readiness thật. Không tự nâng seed/scaffold thành reviewed/verified.</div>
                      {operatingBoard.coverageAudit.truthWarnings.length ? <ul className="mt-3 list-disc space-y-1 pl-5 text-amber-900">{operatingBoard.coverageAudit.truthWarnings.slice(0, 4).map((warning) => <li key={warning}>{warning}</li>)}</ul> : null}
                      {operatingBoard.coverageAudit.priorityActions.length ? (
                        <div className="mt-3 grid gap-2 md:grid-cols-2">
                          {operatingBoard.coverageAudit.priorityActions.slice(0, 4).map((item) => (
                            <div key={`${item.priority}-${item.grade}-${item.action}`} className="rounded-xl border border-amber-200 bg-white p-3">
                              <div className="font-medium text-slate-900">{item.priority.toUpperCase()} · {item.grade}</div>
                              <div className="mt-1 text-slate-700">{item.action}</div>
                              <div className="mt-1 text-xs text-slate-500">{item.reason}</div>
                            </div>
                          ))}
                        </div>
                      ) : null}
                    </div>
                  ) : null}

                  <div className="grid gap-6 xl:grid-cols-[1.1fr,0.9fr]">
                    <div className="rounded-2xl border border-slate-200 bg-white p-4">
                      <div className="mb-3 font-medium text-slate-900">Coverage audit theo lớp</div>
                      <div className="space-y-3">
                        {operatingBoard.coverageAudit.gradeAudits.map((item) => (
                          <div key={item.grade} className="rounded-2xl border border-slate-200 bg-slate-50 p-3 text-sm">
                            <div className="flex flex-wrap items-center gap-2">
                              <span className="font-medium text-slate-900">{item.grade} · {item.level}</span>
                              <Badge>bề mặt {item.surfaceCoveragePercent}%</Badge>
                              <Badge>truth {item.truthReadinessPercent}%</Badge>
                              <Badge>{item.coveredSubjects.length}/{item.requiredSubjects.length} môn</Badge>
                              <Badge>seed {item.seedScopes}</Badge>
                              <Badge>verified {item.verifiedScopes}</Badge>
                              <Badge>operational {item.operationalScopes}</Badge>
                              {item.savedLessons ? <Badge>saved lessons {item.savedLessons}</Badge> : null}
                            </div>
                            {item.missingSubjects.length ? <div className="mt-2 text-xs text-amber-700">Thiếu: {item.missingSubjects.join(', ')}</div> : null}
                            {item.warnings.length ? <div className="mt-2 text-xs text-slate-600">Cảnh báo: {item.warnings.slice(0, 2).join(' ')}</div> : null}
                          </div>
                        ))}
                      </div>
                    </div>

                    <div className="space-y-6">
                      <div className="rounded-2xl border border-slate-200 bg-white p-4">
                        <div className="mb-3 font-medium text-slate-900">Plan/quota foundation</div>
                        <div className="space-y-3">
                          {operatingBoard.plans.map((plan) => (
                            <div key={plan.id} className="rounded-2xl border border-slate-200 bg-slate-50 p-3 text-sm">
                              <div className="font-medium text-slate-900">{plan.name}</div>
                              <div className="mt-1 text-slate-600">{plan.audience}</div>
                              <div className="mt-1 text-xs text-slate-500">{plan.priceNote}</div>
                              <div className="mt-2 flex flex-wrap gap-2 text-xs">
                                <Badge>DOCX {plan.limits.exportDocxPerMonth}/tháng</Badge>
                                <Badge>PDF {plan.limits.exportPdfPerMonth}/tháng</Badge>
                                <Badge>{plan.limits.pdfWatermark ? 'watermark' : 'no watermark'}</Badge>
                              </div>
                            </div>
                          ))}
                        </div>
                      </div>

                      <div className="rounded-2xl border border-slate-200 bg-white p-4">
                        <div className="mb-3 font-medium text-slate-900">Ledger / cộng đồng</div>
                        <div className="grid gap-3 sm:grid-cols-2">
                          <MiniMetric label="Usage records" value={operatingBoard.ledgerSummary.usageRecords} />
                          <MiniMetric label="Point records" value={operatingBoard.ledgerSummary.pointRecords} />
                          <MiniMetric label="Points earned" value={operatingBoard.ledgerSummary.pointsEarned} />
                          <MiniMetric label="Queue kiểm duyệt" value={operatingBoard.communityModeration.moderationQueue} />
                          <MiniMetric label="Activity/Game safe skeleton" value={operatingBoard.activityGameLibrary?.safeSkeletonOnly || 0} />
                          <MiniMetric label="Activity/Game public" value={operatingBoard.activityGameLibrary?.publicAllowed || 0} />
                          <MiniMetric label="Forum queue" value={operatingBoard.forumThreadSafety?.moderationQueue || 0} />
                          <MiniMetric label="Forum reports" value={operatingBoard.forumThreadSafety?.openReports || 0} />
                        </div>
                        <div className="mt-3 rounded-xl border border-slate-200 bg-slate-50 p-3 text-xs text-slate-600">Point policy không có cash equivalent. Community resource public và Activity/Game Library public phải qua moderation, source/license và review; seed chỉ là safe skeleton.</div>
                      </div>
                    </div>
                  </div>


                  <div className="rounded-2xl border border-slate-200 bg-white p-4">
                    <div className="mb-3 flex flex-wrap items-center justify-between gap-2">
                      <div>
                        <div className="font-medium text-slate-900">Forum/Q&A Thread Safety</div>
                        <div className="mt-1 text-sm text-slate-600">Nền tảng hỏi đáp/thảo luận có auto-hold cho thành viên mới, report/takedown và review helpful answer — không public tự do, không AI auto-answer.</div>
                      </div>
                      <div className="flex flex-wrap gap-2">
                        <Badge>new member hold</Badge>
                        <Badge>report auto-hold</Badge>
                        <Badge>no AI answer</Badge>
                      </div>
                    </div>
                    {operatingBoard.forumThreadSafety ? (
                      <div className="space-y-3">
                        <div className="grid gap-3 sm:grid-cols-2 xl:grid-cols-6">
                          <MiniMetric label="Threads" value={operatingBoard.forumThreadSafety.totalThreads} />
                          <MiniMetric label="Public" value={operatingBoard.forumThreadSafety.publicThreads} />
                          <MiniMetric label="Limited" value={operatingBoard.forumThreadSafety.limitedThreads} />
                          <MiniMetric label="Queue" value={operatingBoard.forumThreadSafety.moderationQueue} />
                          <MiniMetric label="Reports" value={operatingBoard.forumThreadSafety.openReports} />
                          <MiniMetric label="Helpful answers" value={operatingBoard.forumThreadSafety.approvedHelpfulAnswers} />
                        </div>
                        <div className="rounded-xl border border-slate-200 bg-slate-50 p-3 text-xs text-slate-600">Forum/Q&A hiện là JSON/source-level foundation. Seed/demo thread không được quảng cáo là cộng đồng thật; thread bị report phải giữ khỏi public cho tới khi triage.</div>
                      </div>
                    ) : <EmptyState label="Chưa có forum thread safety summary." />}
                  </div>

                  <div className="rounded-2xl border border-slate-200 bg-white p-4">
                    <div className="mb-3 flex flex-wrap items-center justify-between gap-2">
                      <div>
                        <div className="font-medium text-slate-900">Community/Creator Hub</div>
                        <div className="mt-1 text-sm text-slate-600">Xếp hạng theo chất lượng, trust level, nguồn/license và đóng góp được duyệt — không xếp theo spam số lượng, không thưởng tiền mặt.</div>
                      </div>
                      <div className="flex flex-wrap gap-2">
                        <Badge>cash reward: tắt</Badge>
                        <Badge>marketplace tiền mặt: tắt</Badge>
                        <Badge>quality leaderboard</Badge>
                      </div>
                    </div>
                    {communityHub ? (
                      <div className="space-y-4">
                        <div className="grid gap-3 sm:grid-cols-2 xl:grid-cols-5">
                          <MiniMetric label="Creator profiles" value={communityHub.summary.creatorProfiles} />
                          <MiniMetric label="Public profiles" value={communityHub.summary.publicCreatorProfiles} />
                          <MiniMetric label="Trusted creators" value={communityHub.summary.trustedCreators} />
                          <MiniMetric label="Queue" value={communityHub.summary.moderationQueue} />
                          <MiniMetric label="Taken down/rejected" value={communityHub.summary.takenDownOrRejected} />
                        </div>
                        {communityHub.leaderboard.length ? (
                          <div className="grid gap-3 md:grid-cols-2">
                            {communityHub.leaderboard.slice(0, 4).map((creator) => (
                              <div key={creator.userId} className="rounded-2xl border border-slate-200 bg-slate-50 p-3 text-sm">
                                <div className="flex flex-wrap items-center gap-2">
                                  <span className="font-medium text-slate-900">{creator.displayName}</span>
                                  <Badge>{creator.trustLevel}</Badge>
                                  <Badge>score {creator.qualityScore}</Badge>
                                </div>
                                <div className="mt-1 text-slate-600">{creator.headline}</div>
                                <div className="mt-2 text-xs text-slate-500">{creator.rankReasons.join(' · ')}</div>
                              </div>
                            ))}
                          </div>
                        ) : <EmptyState label="Public leaderboard đang trống vì chưa có creator đủ điều kiện public. Đây là đúng an toàn khi dữ liệu còn seed/demo." />}
                        {communityHub.warnings.length ? <div className="rounded-xl border border-amber-200 bg-amber-50 p-3 text-xs text-amber-800">{communityHub.warnings.slice(0, 3).join(' ')}</div> : null}
                      </div>
                    ) : <EmptyState label="Chưa tải được Community/Creator Hub." />}
                  </div>

                  <div className="rounded-2xl border border-slate-200 bg-white p-4">
                    <div className="mb-3 font-medium text-slate-900">Validation issues</div>
                    {operatingBoard.issues.length === 0 ? <EmptyState label="Chưa phát hiện blocker/warning ở operating foundation." /> : (
                      <div className="space-y-2">
                        {operatingBoard.issues.map((issue) => (
                          <div key={issue.code} className="rounded-xl border border-slate-200 bg-slate-50 p-3 text-sm">
                            <span className="font-medium text-slate-900">{issue.severity.toUpperCase()} · {issue.code}</span>
                            <div className="mt-1 text-slate-700">{issue.message}</div>

                        ))}
                      </div>
                    )}
                  </div>
                </div>
              ) : <EmptyState label="Chưa tải được operating foundation board." />}
            </Card>
          </div>
        )}

        {activeTab === 'thiet-lap' && (
          <div className="grid gap-6 xl:grid-cols-[1.1fr,0.9fr]">
            <Card title="Thiết lập nhà trường / tổ chuyên môn" desc="Tạo chuẩn dùng chung để thống nhất khi soạn, gửi duyệt và góp ý giáo án.">
              {settingsDraft ? (
                <div className="space-y-4">
                  <Input label="Tên trường" value={settingsDraft.schoolName} onChange={(value) => updateSettingsField('schoolName', value)} />
                  <Input label="Tên tổ / bộ môn" value={settingsDraft.departmentName} onChange={(value) => updateSettingsField('departmentName', value)} />
                  <Input label="Thời lượng mặc định" value={settingsDraft.defaultLessonDuration} onChange={(value) => updateSettingsField('defaultLessonDuration', value)} />
                  <Select label="Cấp học mặc định khi mở workspace" value={settingsDraft.preferredLevel || ''} onChange={(value) => updateSettingsField('preferredLevel', value)} options={settingsLevelOptions} />
                  <Select label="Lớp mặc định khi mở workspace" value={settingsDraft.preferredGrade || ''} onChange={(value) => updateSettingsField('preferredGrade', value)} options={settingsGradeOptions} />
                  <Select label="Môn mặc định khi mở workspace" value={settingsDraft.preferredSubject || ''} onChange={(value) => updateSettingsField('preferredSubject', value)} options={settingsSubjectOptions} />
                  <Select label="Bộ sách mặc định khi mở workspace" value={settingsDraft.preferredBook || ''} onChange={(value) => updateSettingsField('preferredBook', value)} options={settingsBookOptions} />
                  <ToggleRow label="Bắt buộc gửi duyệt trước khi được duyệt chính thức" checked={settingsDraft.approvalRequired} onChange={(value) => updateSettingsField('approvalRequired', value)} />
                  <ToggleRow label="Leader chỉ review/approve trong cùng tổ" checked={settingsDraft.reviewDepartmentOnly} onChange={(value) => updateSettingsField('reviewDepartmentOnly', value)} />
                  <Select label="Phạm vi mặc định cho kho nội dung nội bộ" value={settingsDraft.defaultContentScope} onChange={(value) => updateSettingsField('defaultContentScope', value as any)} options={[{ value: 'department', label: 'Theo tổ / bộ môn' }, { value: 'school', label: 'Toàn trường' }, { value: 'system', label: 'Hệ thống dùng chung' }]} />
                  <Select label="Ngưỡng sourceStatus tối thiểu cho luồng sinh giáo án" value={settingsDraft.teachingMinSourceStatus} onChange={(value) => updateSettingsField('teachingMinSourceStatus', value as any)} options={[{ value: 'seed', label: 'Cho phép seed/demo (phù hợp demo nội bộ)' }, { value: 'reviewed', label: 'Tối thiểu reviewed' }, { value: 'verified', label: 'Tối thiểu verified' }]} />
                  <Select label="Ngưỡng release tier tối thiểu cho luồng sinh giáo án" value={settingsDraft.teachingMinReleaseTier} onChange={(value) => updateSettingsField('teachingMinReleaseTier', value as any)} options={[{ value: 'internal_preview', label: 'internal preview' }, { value: 'school_release_candidate', label: 'school release candidate' }, { value: 'verified_release', label: 'verified release' }]} />
                  <Select label="Ngưỡng support level tối thiểu cho luồng sinh giáo án" value={settingsDraft.teachingMinSupportLevel} onChange={(value) => updateSettingsField('teachingMinSupportLevel', value as any)} options={[{ value: 'starter', label: 'starter' }, { value: 'developing', label: 'developing' }, { value: 'foundation', label: 'foundation' }, { value: 'operational', label: 'operational' }]} />
                  <ToggleRow label="Cho leader/admin bật internal preview bypass khi scope chưa đạt ngưỡng" checked={settingsDraft.allowPrivilegedPreviewBypass} onChange={(value) => updateSettingsField('allowPrivilegedPreviewBypass', value)} />
                  <TextareaField label="Ghi chú dùng chung" value={settingsDraft.sharedNotes} onChange={(value) => updateSettingsField('sharedNotes', value)} />
                  <TextareaField label="Thiết bị / học liệu dùng chung (mỗi dòng một mục)" value={settingsDraft.commonMaterials.join('\n')} onChange={(value) => updateSettingsField('commonMaterials', splitLines(value))} />
                  <TextareaField label="Nguyên tắc góp ý / phản biện (mỗi dòng một mục)" value={settingsDraft.reviewGuidelines.join('\n')} onChange={(value) => updateSettingsField('reviewGuidelines', splitLines(value))} />
                  <TextareaField label="Mẫu giáo án ưu tiên (mỗi dòng một mẫu)" value={settingsDraft.preferredTemplates.join('\n')} onChange={(value) => updateSettingsField('preferredTemplates', splitLines(value))} />
                  <button disabled={!(user?.role === 'leader' || user?.role === 'admin')} onClick={() => void saveSettings()} className="rounded-xl bg-slate-900 px-4 py-2 text-white disabled:cursor-not-allowed disabled:opacity-50">
                    Lưu thiết lập
                  </button>
                </div>
              ) : <EmptyState label="Chưa tải được thiết lập." />}
            </Card>

            <div className="space-y-6">
              <Card title="Thiết lập hiện hành">
                {settings ? (
                  <div className="space-y-4 text-sm text-slate-700">
                    <div className="rounded-2xl border border-slate-200 bg-slate-50 p-4">
                      <div className="font-medium text-slate-900">{settings.schoolName}</div>
                      <div className="mt-1 text-slate-600">{settings.departmentName}</div>
                      <div className="mt-2 text-slate-600">Thời lượng mặc định: {settings.defaultLessonDuration}</div>
                      <div className="mt-1 text-slate-600">Quy trình duyệt: {settings.approvalRequired ? 'Bắt buộc gửi duyệt' : 'Cho phép lưu và duyệt ngay'}</div>
                      <div className="mt-1 text-slate-600">Phạm vi review của leader: {settings.reviewDepartmentOnly ? 'Cùng tổ' : 'Toàn trường'}</div>
                      <div className="mt-1 text-slate-600">Scope mặc định kho nội dung: {settings.defaultContentScope}</div>
                      <div className="mt-1 text-slate-600">Mở workspace mặc định: {settings.preferredLevel} · {settings.preferredGrade} · {settings.preferredSubject} · {settings.preferredBook}</div>
                      <div className="mt-1 text-slate-600">Ngưỡng sinh giáo án: {settings.teachingMinSourceStatus} · {settings.teachingMinReleaseTier} · {labelCurriculumSupportLevel(settings.teachingMinSupportLevel)}</div>
                      <div className="mt-1 text-slate-600">Preview bypass cho leader/admin: {settings.allowPrivilegedPreviewBypass ? 'Cho phép' : 'Tắt'}</div>
                    </div>
                    <InfoList title="Mẫu ưu tiên" items={settings.preferredTemplates} compact />
                    <InfoList title="Thiết bị / học liệu dùng chung" items={settings.commonMaterials} compact />
                    <InfoList title="Nguyên tắc góp ý" items={settings.reviewGuidelines} compact />
                  </div>
                ) : <EmptyState label="Chưa có thiết lập hiện hành." />}
              </Card>

              <Card title="Kho nội dung dùng cho workspace" desc="Đồng bộ seed có cấu trúc vào JSON store hoặc database để chuẩn bị cho giai đoạn nhập liệu chuẩn thật.">
                <div className="space-y-4 text-sm text-slate-700">
                  <div className="rounded-2xl border border-slate-200 bg-slate-50 p-4">
                    <div className="font-medium text-slate-900">Nguồn hiện hành: {labelContentMode(metadata?.contentOverview?.sourceMode)}</div>
                    <div className="mt-1 text-slate-600">Chủ đề {metadata?.contentOverview?.topics ?? 0} · Câu hỏi {metadata?.contentOverview?.questions ?? 0} · Rubric {metadata?.contentOverview?.rubrics ?? 0}</div>
                  </div>
                  <div className="flex flex-wrap gap-3">
                    <button disabled={!(user?.role === 'leader' || user?.role === 'admin') || syncingContent} onClick={() => void syncContent('auto')} className="rounded-xl bg-slate-900 px-4 py-2 text-white disabled:cursor-not-allowed disabled:opacity-50">Đồng bộ vào nguồn hiện hành</button>
                    <button disabled={!(user?.role === 'leader' || user?.role === 'admin') || syncingContent} onClick={() => void syncContent('json')} className="rounded-xl border border-slate-300 bg-white px-4 py-2 disabled:cursor-not-allowed disabled:opacity-50">Đồng bộ JSON store</button>
                    <button disabled={!(user?.role === 'admin') || syncingContent} onClick={() => void syncContent('database')} className="rounded-xl border border-slate-300 bg-white px-4 py-2 disabled:cursor-not-allowed disabled:opacity-50">Đồng bộ database</button>
                  </div>
                  {contentSyncMessage ? <div className="rounded-2xl border border-slate-200 bg-slate-50 px-4 py-3 text-sm text-slate-700">{contentSyncMessage}</div> : null}
                </div>
              </Card>

              <Card title="Quyền cập nhật">
                <div className="space-y-2 text-sm text-slate-700">
                  <div>Giáo viên: xem thiết lập và dùng để soạn giáo án.</div>
                  <div>Tổ trưởng / chuyên môn: cập nhật thiết lập, góp ý, duyệt giáo án, đồng bộ JSON store.</div>
                  <div>Quản trị: toàn quyền cập nhật, điều phối hệ thống và đồng bộ database.</div>
                </div>
              </Card>
            </div>
          </div>
        )}

        {activeTab === 'quan-tri-noi-dung' && (
          <ContentAdminPanel
            user={user}
            level={level}
            grade={grade}
            subject={subject}
            book={book}
            topic={topic}
            onChanged={() => {
              void loadBoot();
              void loadProgram();
              void loadBanks();
              void loadResources(resourceSearch);
            }}
          />
        )}

      </div>
    </main>
  );
}

function labelSourceStatus(status?: 'seed' | 'reviewed' | 'verified') {
  if (status === 'verified') return 'verified';
  if (status === 'reviewed') return 'reviewed';
  return 'seed/demo';
}

function labelCurriculumSupportLevel(level?: CurriculumSupportLevel) {
  if (level === 'operational') return 'vận hành';
  if (level === 'foundation') return 'foundation';
  if (level === 'developing') return 'đang làm dày';
  return 'starter';
}

function labelWorkflowAction(action: WorkflowEvent['action']) {

function formatDate(value: string) {
  try {
    return new Intl.DateTimeFormat('vi-VN', { dateStyle: 'short', timeStyle: 'short' }).format(new Date(value));
  } catch {
    return value;
  }
}

function splitLines(value: string) {
  return value.split('\n').map((item) => item.trim()).filter(Boolean);
}

function labelLessonQualityDecision(decision?: LessonQualityChecklistReport['summary']['decision']) {
  if (decision === 'release_blocked') return 'Chặn gửi duyệt/chính thức';
  if (decision === 'needs_revision') return 'Cần sửa thêm';
  if (decision === 'editable_draft') return 'Bản nháp có cảnh báo';
  return 'Ứng viên gửi duyệt';
}

function LessonQualityPanel({ report }: { report?: LessonQualityChecklistReport | null }) {
  if (!report) {
    return <div className="mb-3 rounded-2xl border border-slate-200 bg-slate-50 px-4 py-3 text-sm text-slate-600">Chưa có checklist chất lượng cho nội dung đang soạn. Bấm “Checklist giáo án” hoặc sinh giáo án mới để kiểm tra rule-based.</div>;
  }
  const topIssues = report.issues.filter((issue) => issue.severity === 'blocker' || issue.severity === 'warning').slice(0, 5);
  return (
    <div className="mb-3 rounded-2xl border border-slate-200 bg-slate-50 px-4 py-3 text-sm">
      <div className="flex flex-wrap items-center gap-2">
        <span className="font-medium text-slate-900">Checklist chất lượng giáo án</span>
        <Badge>{report.summary.score}/100</Badge>
        <Badge>{labelLessonQualityDecision(report.summary.decision)}</Badge>
        <Badge>{report.gradeBand}</Badge>
        <Badge>blocker {report.summary.blockers}</Badge>
        <Badge>cảnh báo {report.summary.warnings}</Badge>
      </div>
      <div className="mt-2 text-slate-600">Đạt {report.summary.passed}/{report.summary.total} tiêu chí; tiêu chí bắt buộc {report.summary.requiredPassed}/{report.summary.requiredTotal}. Đây là checklist rule-based, không thay review chuyên môn.</div>
      {topIssues.length > 0 ? (
        <ul className="mt-2 list-disc space-y-1 pl-5 text-xs text-amber-800">
          {topIssues.map((issue) => <li key={issue.code}>{issue.message}</li>)}
        </ul>
      ) : <div className="mt-2 text-xs text-emerald-700">Chưa thấy blocker/cảnh báo lớn theo rule hiện tại.</div>}
    </div>
  );
}

function LessonGovernancePanel({ snapshot, settings, compact = false, title = 'Governance / provenance' }: { snapshot?: LessonGovernanceSnapshot | null; settings?: SchoolSettings | null; compact?: boolean; title?: string }) {
  if (!snapshot) {
    return <div className={`rounded-2xl border border-amber-200 bg-amber-50 px-4 py-3 text-sm text-amber-800 ${compact ? 'mt-2' : 'mb-3'}`}>Chưa có snapshot governance/provenance được lưu cùng giáo án này.</div>;
  }
  const assessment = settings ? assessLessonGovernanceForApproval({ snapshot, settings }) : null;
  const effective = assessment?.effectiveSnapshot || normalizeLessonGovernanceSnapshot(snapshot);
  const blockers = assessment?.issues.filter((item) => item.severity === 'blocker') || [];
  const warnings = assessment?.issues.filter((item) => item.severity === 'warning') || [];
  return (
    <div className={`rounded-2xl border border-slate-200 bg-slate-50 px-4 py-3 text-sm ${compact ? 'mt-2' : 'mb-3'}`}>
      <div className="flex flex-wrap items-center gap-2">
        <span className="font-medium text-slate-900">{title}</span>
        <Badge>{labelSourceStatus(effective.sourceStatus)}</Badge>
        <Badge>{effective.releaseTier}</Badge>
        <Badge>{labelCurriculumSupportLevel(effective.supportLevel)}</Badge>
        {assessment ? <Badge>{labelLessonGovernanceApproval(assessment)}</Badge> : null}
        <Badge>{effective.sourceReferences.length || effective.referenceCount || 0} ref</Badge>
      </div>
      <div className="mt-2 text-slate-600">{effective.packId ? `Pack ${effective.packId}` : 'Chưa gắn pack'} · {effective.sourceLabel}</div>
      {blockers.length > 0 ? <div className="mt-2 rounded-xl border border-rose-200 bg-rose-50 px-3 py-2 text-xs text-rose-800">Blocker: {blockers.map((item) => item.message).join(' ')}</div> : null}
      {blockers.length === 0 && warnings.length > 0 ? <div className="mt-2 rounded-xl border border-amber-200 bg-amber-50 px-3 py-2 text-xs text-amber-800">Cảnh báo: {warnings.slice(0, 2).map((item) => item.message).join(' ')}</div> : null}
    </div>
  );
}

function BarList({ items }: { items: Array<{ name: string; value: number }> }) {
  const max = Math.max(...items.map((item) => item.value), 1);
  if (!items.length) return <EmptyState label="Chưa có dữ liệu." />;
  return <div className="space-y-3">{items.map((item) => <div key={item.name}><div className="mb-1 flex items-center justify-between text-sm text-slate-700"><span>{item.name}</span><strong>{item.value}</strong></div><div className="h-2 rounded-full bg-slate-100"><div className="h-2 rounded-full bg-slate-900" style={{ width: `${(item.value / max) * 100}%` }} /></div></div>)}</div>;
}

function EmptyState({ label }: { label: string }) {
  return <div className="rounded-2xl border border-dashed border-slate-300 bg-slate-50 px-4 py-6 text-center text-sm text-slate-500">{label}</div>;
}
